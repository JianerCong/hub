  let c1, a
  c1 = document.getElementsByClassName("TOCTitle--fFPXo");
  c1.length
  a = ""
  for (let h of c1){
      s = h.firstElementChild.innerText
      s = "+ " + s
      for (let i=1;i<s.split('.').length-1;i++){// indent the list
          s = "  " + s
      }
      a += s + "\n"
      //console.log(h.firstElementChild.innerHTML)

  }
  console.log(a)
