from pathlib import Path

def test_basic():
    p = Path.home()
    p1 = Path.cwd()
    assert p.exists()
    assert p1.exists()
    assert p.is_dir()
    assert not p.is_file()
    assert p.is_absolute()


def test_mkdir():
    p = Path.cwd() / 'd1'
    assert not p.exists()
    assert not p.is_file()
    assert not p.is_dir()

    p.mkdir()

    assert p.exists()
    assert p.is_dir()

    p.rmdir()                   # must be empty
    assert not p.exists()

def test_write_txt():
    p = Path.cwd() / 'hi.txt'
    assert not p.exists()
    assert not p.is_file()
    assert not p.is_dir()


    assert 3 == p.write_text('abc')
    assert p.exists()
    assert p.is_file()
    assert p.read_text() == 'abc'

    p.unlink()                   # remove the file
    assert not p.exists()
